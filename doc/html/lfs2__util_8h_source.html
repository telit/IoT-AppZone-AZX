152 // Find the smallest power of 2 greater than or equal to a
153 static inline uint32_t lfs2_npw2(uint32_t a) {
154 #if !defined(LFS2_NO_INTRINSICS) && MBED_LFS2_INTRINSICS && \
155  (defined(__GNUC__) || defined(__CC_ARM))
156  return 32 - __builtin_clz(a-1);
157 #else
158  uint32_t r = 0;
159  uint32_t s;
160  a -= 1;
161  s = (a > 0xffff) << 4; a >>= s; r |= s;
162  s = (a > 0xff ) << 3; a >>= s; r |= s;
163  s = (a > 0xf ) << 2; a >>= s; r |= s;
164  s = (a > 0x3 ) << 1; a >>= s; r |= s;
165  return (r | (a >> 1)) + 1;
166 #endif
167 }

169 // Count the number of trailing binary zeros in a
170 // lfs2_ctz(0) may be undefined
171 static inline uint32_t lfs2_ctz(uint32_t a) {
172 #if !defined(LFS2_NO_INTRINSICS) && MBED_LFS2_INTRINSICS && \

181 static inline uint32_t lfs2_popc(uint32_t a) {
182 #if !defined(LFS2_NO_INTRINSICS) && MBED_LFS2_INTRINSICS && \
183  (defined(__GNUC__) || defined(__CC_ARM))

198 // Convert between 32-bit little-endian and native order
199 static inline uint32_t lfs2_fromle32(uint32_t a) {
200 #if !defined(LFS2_NO_INTRINSICS) && MBED_LFS2_INTRINSICS && ( \
201  (defined( BYTE_ORDER ) && defined( ORDER_LITTLE_ENDIAN ) && BYTE_ORDER == ORDER_LITTLE_ENDIAN ) || \
202  (defined(__BYTE_ORDER ) && defined(__ORDER_LITTLE_ENDIAN ) && __BYTE_ORDER == __ORDER_LITTLE_ENDIAN ) || \
203  (defined(__BYTE_ORDER__) && defined(__ORDER_LITTLE_ENDIAN__) && __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__))
204  return a;
205 #elif !defined(LFS2_NO_INTRINSICS) && MBED_LFS2_INTRINSICS && ( \
206  (defined( BYTE_ORDER ) && defined( ORDER_BIG_ENDIAN ) && BYTE_ORDER == ORDER_BIG_ENDIAN ) || \
207  (defined(__BYTE_ORDER ) && defined(__ORDER_BIG_ENDIAN ) && __BYTE_ORDER == __ORDER_BIG_ENDIAN ) || \
208  (defined(__BYTE_ORDER__) && defined(__ORDER_BIG_ENDIAN__) && __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__))
209  return __builtin_bswap32(a);
210 #else
211  return (((uint8_t*)&a)[0] << 0) |
212  (((uint8_t*)&a)[1] << 8) |
213  (((uint8_t*)&a)[2] << 16) |
214  (((uint8_t*)&a)[3] << 24);
215 #endif
216 }

222 // Reverse the bits in a
223 static inline uint32_t lfs2_rbit(uint32_t a) {
224 #if !defined(LFS2_NO_INTRINSICS) && MBED_LFS2_INTRINSICS && \
225  defined(__MBED__)
226  return __RBIT(a);
227 #else
228  a = ((a & 0xaaaaaaaa) >> 1) | ((a & 0x55555555) << 1);
229  a = ((a & 0xcccccccc) >> 2) | ((a & 0x33333333) << 2);
230  a = ((a & 0xf0f0f0f0) >> 4) | ((a & 0x0f0f0f0f) << 4);
231  a = ((a & 0xff00ff00) >> 8) | ((a & 0x00ff00ff) << 8);
232  a = (a >> 16) | (a << 16);
233  return a;
234 #endif
235 }
